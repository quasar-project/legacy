66 QString ColorProvider::next() const
67 {
68 QString ret;
69 switch(m_currentBehavior)
70 {
71 case OrderedCycle: {
72 switch(m_currentTheme)
73 {
74 case Light: ret = m_light_palette.at((BaseColor)m_it.count).name(); break;
75 case Dark: ret = m_dark_palette.at((BaseColor)m_it.count).name(); break;
76 }
77
78 ++m_it.count;
79 if(m_it.count > Max) {
80 m_it.count = Min;
81 ++m_it.loop;
82 }
83 break;
84 }
85 case RandomCycle: {
86 int current_idx = random_in_range(Min, Max);
87
88 switch(m_currentTheme)
89 {
90 case Light: ret = m_light_palette.at((BaseColor)current_idx).name(); break;
91 case Dark: ret = m_dark_palette.at((BaseColor)current_idx).name(); break;
92 }
93
94 ++m_it.count;
95 if(m_it.count > Max) {
96 m_it.count = Min;
97 ++m_it.loop;
98 }
99 break;
100 }
101 case OrderedTint: {
102 switch(m_currentTheme)
103 {
104 case Light: ret = m_light_palette.at((BaseColor)m_it.count).lighter(100 + m_it.loop * 10).name(); break;
105 case Dark: ret = m_dark_palette.at((BaseColor)m_it.count).darker(100 + m_it.loop * 10).name(); break;
106 }
107
108 ++m_it.count;
109 if(m_it.count > Max) {
110 m_it.count = Min;
111 ++m_it.loop;
112 }
113 break;
114 }
115 case RandomTint: {
116 int current_idx = random_in_range(Min, Max);
117 bool tint = false;
118 if(m_exclude_list[current_idx])
119 tint = true;
120
121 switch(m_currentTheme)
122 {
123 case Light: ret = m_light_palette.at((BaseColor)current_idx).lighter(100 + (tint ? m_it.loop * 20 : m_it.loop * 10)).name(); break;
124 case Dark: ret = m_dark_palette.at((BaseColor)current_idx).darker(100 + (tint ? m_it.loop * 20 : m_it.loop * 10)).name(); break;
125 }
126
127 ++m_it.count;
128 m_exclude_list[current_idx] = true;
129 if(m_it.count > Max) {
130 m_it.count = Min;
131 ++m_it.loop;
132
133 for(auto& state : m_exclude_list)
134 state = false;
135 }
136 break;
137 }
138 }
139
140 return ret;
141 }
