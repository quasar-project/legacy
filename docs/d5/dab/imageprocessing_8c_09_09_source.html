84void ImageProcessing::asyncProcess(const QString& filename)
85{
86 Map::TelescopicImage image;
87 try
88 {
89 image = decodeTelescopic(Config::Paths::lod(0) + "/" + filename);
90 }
91 catch(const std::runtime_error& err)
92 {
93 qCritical() << "[PROCESSING] Caught exception";
94 qCritical() << err.what();
95 return;
96 }
97
98 image.filename = filename;
99 image.opacity = ICFG<float>("PROCESSING_IMAGE_INITIAL_OPACITY");
100 image.shown = ICFG<bool>("PROCESSING_IMAGE_INITIAL_VISIBILITY");
101 image.mercator_zoom_level = SDK::Cartography::mqi_zoom_level(image.meta.latitude, image.meta.dx);
102
103 QImage image_data(image.path.first);
104 if(image_data.isNull()) {
105 qCritical() << "[PROCESSING] Bad image file";
106 }
107
108 image.meta.ly = static_cast<float>(image_data.height());
109
110 QString target_filename = filename;
111 target_filename.chop(3);
112 target_filename += "png";
113
114 if(Config::Settings::get()->value<bool>("image/cut") and image.meta.image_type == 0)
115 image_data = cutImage(image);
116
117 bool ret = image_data.save(Config::Paths::lod(0) + "/" + target_filename);
118 if(ret)
119 qInfo() << "[PROCESSING] Image saved successfully";
120 else
121 qCritical() << "[PROCESSING] Failed to save image";
122
123 image.path.first = Config::Paths::lod(0) + "/" + target_filename;
124 QFile::remove(Config::Paths::lod(0) + "/" + filename);
125
126 emit concurrencyFinished();
127 emit processImageFinished(image);
128}

142Map::TelescopicImage ImageProcessing::decodeTelescopic(const QString& path)
143{
144 Map::TelescopicImage image;
145 image.path.first = path;
146 QByteArray data = fileToByteArray(image.path.first);
147
148 char* data_ptr = data.data();
149 auto* marker = reinterpret_cast<uint16_t*>(data_ptr + image.header.JPEG_HEADER_SIZE);
150
151 if(*marker != image.header.meta_marker and *marker != qFromBigEndian<uint16_t>(image.header.meta_marker))
152 throw std::runtime_error("[PROCESSING] Marker mismatch exception");
153
154 auto* meta_size = reinterpret_cast<uint16_t*>(data_ptr + image.header.JPEG_HEADER_SIZE + sizeof(uint16_t));
155 *meta_size = qToBigEndian(*meta_size) - sizeof(uint16_t);
156 memcpy(&image.meta, (data_ptr + image.header.JPEG_HEADER_SIZE + sizeof(uint32_t)), *meta_size);
157
158 char* crc_data = (char*)&image.meta;
159 uint16_t crc16 = SDK::Crypto::crc16(crc_data, sizeof(Map::ImageMetadata) - sizeof(uint16_t));
160
161 image.valid = crc16 == image.meta.crc16;
162 if(not image.valid)
163 qWarning().noquote().nospace() << "[PROCESSING] Image CRC16 seems to be incorrect: expected 0x"
164 << Qt::hex << crc16 << ", received 0x" << image.meta.crc16 << Qt::dec;
165
166 if(Config::Settings::get()->value<bool>("image/radians"))
167 {
168 image.meta.angle = SDK::rad2deg(image.meta.angle);
169 image.meta.drift_angle = Config::Settings::get()->value<bool>("image/drift-angle")
170 ? SDK::rad2deg(image.meta.drift_angle) : 0;
171 image.meta.div = SDK::rad2deg(image.meta.div);
172 };
173
174 if(qIsNaN(image.meta.drift_angle))
175 image.meta.drift_angle = 0;
176
177 return image;
178}

180QImage ImageProcessing::cutImage(const Map::TelescopicImage& image) noexcept
181{
182 QImage image_data(image.path.first);
183 QImage sample(image_data.width(), image_data.height(), QImage::Format_ARGB32_Premultiplied);
184 sample.fill(QColor(254, 254, 254));
185 image_data.setAlphaChannel(sample);
186
187 QPainter painter;
188 QPainterPath path;
189 QBrush brush(Qt::transparent, Qt::SolidPattern);
190 QPolygon p1, p2;
191
192 brush.setStyle(Qt::SolidPattern);
193 brush.setColor(Qt::transparent);
194 painter.begin(&image_data);
195 painter.setCompositionMode(QPainter::CompositionMode_Source);
196 painter.setPen(QPen(Qt::transparent, 10));
197 painter.setRenderHint(QPainter::Antialiasing);
198
199 const int top[8] =
200 {
201 0, (int)((image.meta.ly / 2) - 2 * image.meta.x0 * tan(SDK::deg2rad((image.meta.div - Config::Settings::get()->value<float>("image/div-correction")) / 2))),
202 0, 0,
203 (int)image.meta.lx, 0,
204 (int)image.meta.lx, (int)((image.meta.ly / 2) - (2 * image.meta.x0 + image.meta.lx) * tan(SDK::deg2rad((image.meta.div - Config::Settings::get()->value<float>("image/div-correction")) / 2)))
205 };
206
207 const int bottom[8] =
208 {
209 0, (int)((image.meta.ly / 2) + 2 * image.meta.x0 * tan(SDK::deg2rad((image.meta.div - Config::Settings::get()->value<float>("image/div-correction")) / 2))),
210 0, (int)image.meta.ly,
211 (int)image.meta.lx, (int)image.meta.ly,
212 (int)image.meta.lx, (int)((image.meta.ly / 2) + (2 * image.meta.x0 + image.meta.lx) * tan(SDK::deg2rad((image.meta.div - Config::Settings::get()->value<float>("image/div-correction")) / 2)))
213 };
214
215 p1.setPoints(4, top);
216 p2.setPoints(4, bottom);
217 painter.drawPolygon(p1);
218 painter.drawPolygon(p2);
219 path.addPolygon(p1);
220 path.addPolygon(p2);
221
222 painter.fillPath(path, brush);
223 painter.end();
224
225 return image_data;
226}
