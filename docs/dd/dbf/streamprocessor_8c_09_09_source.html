46 void StreamProcessor::processChunk() noexcept
47 {
48 if(m_queue.empty())
49 {
50 emit queueFinished();
51 return;
52 }
53
54 QByteArray chunk = m_queue.front();
55 m_queue.pop();
56
57 const int MAX_PACKAGE_SIZE = ICFG<int>("PROCESSING_STRIP_PACKAGE_MAX_SIZE");
58
59 MapImageSegment image;
60
61 const uint8_t* data = reinterpret_cast<uint8_t*>(chunk.data());
62 int data_size = chunk.size();
63
64 const int header_size = sizeof(Datagrams::StripImageChunkHeader);
65 uint8_t header[header_size];
66 uint8_t buf[MAX_PACKAGE_SIZE - header_size];
67 float fbuf[MAX_PACKAGE_SIZE - header_size];
68
69 ArrayReader<uint8_t> array_reader((uint8_t*)data);
70
71 int x = 0;
72 int y = 0;
73 vector<float> fmatrix;
74
75 bool navigation_read = false;
76 while(array_reader.readed() < data_size)
77 {
78 if((data_size - array_reader.readed()) < header_size)
79 break;
80
81 array_reader.read(header, header_size);
82 ArrayReader<uint8_t>header_reader(header);
83
84 Datagrams::StripImageChunkID head;
85 Datagrams::StripImageChunkNav nav;
86 Datagrams::StripImageChunkFormat img;
87
88 header_reader.read((uint8_t*)&head, sizeof(Datagrams::StripImageChunkID));
89
90 if(head.marker != 0xDEFA and head.marker != 0xFADE)
91 {
92 if(not m_queue.empty())
93 emit queuePassed();
94 return;
95 }
96
97 header_reader.read((uint8_t*)&nav, sizeof(Datagrams::StripImageChunkNav));
98 header_reader.read((uint8_t*)&img, sizeof(Datagrams::StripImageChunkFormat));
99
100 if(head.cnt == 0)
101 {
102 x = img.nx;
103 y += img.ny;
104 }
105
106 if(not navigation_read)
107 {
108 image.setCoordinate(Utils::rad2deg(nav.latitude), Utils::rad2deg(nav.longitude));
109 image.setAzimuth(static_cast<float>(Utils::rad2deg(nav.course)));
110 qCritical() << "\t Пришло:" << nav.course;
111 qCritical() << "\t Конверсия в градусы:" << image.azimuth();
112 image.setRectSize(QSizeF(img.nx, img.ny));
113 image.setRatio(img.dx);
114 float _x_ = -img.x0;
115 float _y_ = static_cast<float>(img.y) / 10.0f;
116 const float alpha = Utils::deg2rad(image.azimuth());
117 qCritical() << "\t Конверсия для матрицы:" << alpha;
118 image.setOffset(QPointF(_x_ * std::cos(alpha) - _y_ * std::sin(alpha),
119 _x_ * std::sin(alpha) + _y_ * std::cos(alpha)));
120 navigation_read = true;
121 qCritical() << "\t Передано в QML:" << image.azimuth();
122 }
123
124 array_reader.read((uint8_t*)buf, head.size);
125
126 // запись промежуточного результата в матрицу
127 for(int i = 0; i < head.size; i++)
128 {
129 if(i >= MAX_PACKAGE_SIZE - header_size)
130 break;
131 auto f = static_cast<float>(static_cast<float>(buf[i]) * img.k);
132 fmatrix.push_back(f);
133 }
134 }
135
136 // Эти значения нужно пересчитывать каждый раз при выводе матрицы на экран
137 // Лучше сделать вычисление максимального значения рекуррентно
138 const float max_value = *max_element(fmatrix.begin(), fmatrix.end());
139 const float k = max_value / 255.0f;
140
141 int out_size = x * y;
142 uint8_t out_buf[out_size];
143
144 // обратное масштабирование
145 for(int i = 0; i < out_size; i++)
146 out_buf[i] = static_cast<uint8_t>(fmatrix[i] / k);
147
148 vector<uint8_t> output(out_buf, out_buf + out_size);
149
150 QImage strip_result(x, y, QImage::Format_Grayscale16);
151 vector<vector<float>> image_result;
152 image_result.resize(y);
153 for(size_t i = 0; i < y; ++i)
154 {
155 image_result[i].resize(x);
156 for(size_t j = 0; j < x; ++j)
157 image_result[i][j] = output[j + x * i];
158 }
159 //strip_result = strip_result.mirrored(true, true);
160
161 for(size_t row = 0; row < image_result.size(); ++row)
162 {
163 for(size_t column = 0; column < image_result[row].size(); ++column)
164 {
165 float t = (image_result[row][column]);
166 uint8_t d[4] = { static_cast<uint8_t>(t), static_cast<uint8_t>(t), static_cast<uint8_t>(t), 255 };
167 auto* rs = (uint32_t*)d;
168 strip_result.setPixel(column, row, *rs);
169 }
170 }
171
172 QByteArray temporary;
173 QBuffer buffer(&temporary);
174 buffer.open(QIODevice::WriteOnly);
175 strip_result.save(&buffer, "PNG");
176
177 image.setData(temporary);
178 image.setVisible(ICFG<bool>("PROCESSING_IMAGE_INITIAL_VISIBILITY"));
179 image.setOpacity(ICFG<float>("PROCESSING_IMAGE_INITIAL_OPACITY"));
180
181 emit finished(image);
182 if(not m_queue.empty())
183 emit queuePassed();
184 }
