68 void ExecdSocket::execute(Enums::NetworkCommand command) noexcept
69 {
70 QByteArray com;
71
72 switch(command)
73 {
74 case Enums::FormTelescopic:
75 com = Utils::wrapToExecdString(FROM_JSON("EXECD_FORM_TELESCOPIC") + m_args->formArgumentString(), &m_message_uid);
76 break;
77 case Enums::FocusImage:
78 com = Utils::wrapToExecdString(FROM_JSON("EXECD_FOCUS_TELESCOPIC") + m_args->focusArgumentString(), &m_message_uid);
79 break;
80 case Enums::ReformImage:
81 com = Utils::wrapToExecdString(FROM_JSON("EXECD_FORM_TELESCOPIC") + m_args->reformArgumentString(), &m_message_uid);
82 break;
83 case Enums::RemoteStorageStatus:
84 com = Utils::wrapToExecdString(FROM_JSON("EXECD_STORAGE_FETCH"), &m_message_uid);
85 break;
86 case Enums::ClearRemoteStorage:
87 com = Utils::wrapToExecdString(FROM_JSON("EXECD_STORAGE_CLEAR") + " " + FROM_JSON("EXECD_STORAGE_FETCH"), &m_message_uid);
88 break;
89 case Enums::StripStart:
90 {
91 if(m_compatibilityMode)
92 com = Utils::wrapToExecdString(FROM_JSON("EXECD_COMPAT_STRIP_START") + m_args->formArgumentString(), &m_message_uid);
93 else
94 {
95 QString strip_comma = FROM_JSON("EXECD_FORM_STRIP_START") + m_args->formArgumentString();
96 com = Utils::wrapToExecdString(condition(FROM_JSON("EXECD_SPECIAL_PID_OF") + "(strip_shot)",
97 FROM_JSON("EXECD_SPECIAL_PASS"), strip_comma), &m_message_uid);
98 m_strip_pid = m_message_uid;
99 }
100 break;
101 }
102 case Enums::StripStop:
103 if(m_compatibilityMode)
104 {
105 com = Utils::wrapToExecdString(FROM_JSON("EXECD_COMPAT_STRIP_STOP") + m_args->formArgumentString(), &m_message_uid);
106 break;
107 }
108 else
109 {
110 signalToProcess(m_strip_pid, Enums::SigINT);
111 return;
112 }
113 case Enums::StreamStart:
114 {
115 if(m_compatibilityMode)
116 return;
117 QString stream_comma = FROM_JSON("EXECD_FORM_STREAM_START") + m_args->formArgumentString();
118 com = Utils::wrapToExecdString(condition(FROM_JSON("EXECD_SPECIAL_PID_OF") + "(strip)",
119 FROM_JSON("EXECD_SPECIAL_PASS"), stream_comma), &m_message_uid);
120 m_strip_pid = m_message_uid;
121 break;
122 }
123 case Enums::StreamStop:
124 if(m_compatibilityMode)
125 return;
126 signalToProcess(m_strip_pid, Enums::SigINT);
127 return;
128 case Enums::Reboot:
129 {
130 com = Utils::wrapToExecdString(FROM_JSON("EXECD_REBOOT"), &m_message_uid);
131 break;
132 }
133 case Enums::PowerOff:
134 {
135 com = Utils::wrapToExecdString(FROM_JSON("EXECD_POWEROFF"), &m_message_uid);
136 break;
137 }
138 default:
139 qWarning() << "[EXECD] Incorrect command type";
140 return;
141 }
142
143 this->send(com);
144
145 qDebug().noquote() << "[EXECD] Sent built-in command";
146 emit metrics(com, com.length(), true);
147 }

151 void ExecdSocket::process(QByteArray data)
152 {
153 QString raw = data.data();
154 QString check_crc = raw;
155 check_crc.chop(5);
156
157 uint16_t crc16 = Utils::crc16(Utils::str_data(check_crc), check_crc.length());
158 uint16_t receivedCrc16 = raw.split("|").last().toUInt(nullptr, 16);
159
160 if(crc16 == receivedCrc16)
161 {
162 qDebug() << "[EXECD] Command executed successfully";
163 emit ping();
164 }
165// else
166// qWarning() << "[EXECD] CRC16 mismatch";
167 emit metrics(raw, data.size(), false);
168 }
