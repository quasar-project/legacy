57 void ExecdSocket::execute(Enums::NetworkCommand command) noexcept
58 {
59 QByteArray com;
60
61 switch(command)
62 {
63 case Enums::FormTelescopic:
64 com = wrap(FROM_JSON("EXECD_FORM_TELESCOPIC") + m_args->formArgumentString());
65 break;
66 case Enums::FocusImage:
67 com = wrap(FROM_JSON("EXECD_FOCUS_TELESCOPIC") + m_args->focusArgumentString());
68 break;
69 case Enums::ReformImage:
70 com = wrap(FROM_JSON("EXECD_FORM_TELESCOPIC") + m_args->reformArgumentString());
71 break;
72 case Enums::RemoteStorageStatus:
73 com = wrap(FROM_JSON("EXECD_STORAGE_FETCH"));
74 break;
75 case Enums::ClearRemoteStorage:
76 com = wrap(FROM_JSON("EXECD_STORAGE_CLEAR") + " " + FROM_JSON("EXECD_STORAGE_FETCH"));
77 break;
78 case Enums::StripStart:
79 {
80 QString strip_comma = FROM_JSON("EXECD_FORM_STRIP_START") + m_args->formArgumentString();
81 com = wrap(condition(FROM_JSON("EXECD_SPECIAL_PID_OF") + "(strip_shot)",
82 FROM_JSON("EXECD_SPECIAL_PASS"), strip_comma));
83 m_strip_pid = m_message_uid;
84 break;
85 }
86 case Enums::StripStop:
87 signalToProcess(m_strip_pid, Enums::SigINT);
88 return;
89 case Enums::StreamStart:
90 {
91 QString stream_comma = FROM_JSON("EXECD_FORM_STREAM_START") + m_args->formArgumentString();
92 com = wrap(condition(FROM_JSON("EXECD_SPECIAL_PID_OF") + "(strip)",
93 FROM_JSON("EXECD_SPECIAL_PASS"), stream_comma));
94 m_strip_pid = m_message_uid;
95 break;
96 }
97 case Enums::StreamStop:
98 signalToProcess(m_strip_pid, Enums::SigINT);
99 return;
100 case Enums::Reboot:
101 {
102 com = wrap(FROM_JSON("EXECD_REBOOT"));
103 break;
104 }
105 case Enums::PowerOff:
106 {
107 com = wrap(FROM_JSON("EXECD_POWEROFF"));
108 break;
109 }
110 default:
111 qWarning() << "[EXECD] Incorrect command type";
112 return;
113 }
114
115 this->send(com);
116
117 qDebug().noquote() << "[EXECD] Sent built-in command";
118 emit metrics(com, com.length(), true);
119 }

123 void ExecdSocket::process(QByteArray data)
124 {
125 QString raw = data.data();
126 QString check_crc = raw;
127 check_crc.chop(5);
128
129 uint16_t crc16 = Utils::crc16(Utils::str_data(check_crc), check_crc.length());
130 uint16_t receivedCrc16 = raw.split("|").last().toUInt(nullptr, 16);
131
132 if(crc16 == receivedCrc16)
133 {
134 qDebug() << "[EXECD] Command executed successfully";
135 emit ping();
136 }
137// else
138// qWarning() << "[EXECD] CRC16 mismatch";
139 emit metrics(raw, data.size(), false);
140 }

151 QByteArray ExecdSocket::wrap(const QString& string) noexcept
152 {
153 // prevents UID overflow
154 if(m_message_uid >= 9999)
155 m_message_uid = 0;
156
157 QString command = ":" + QStringLiteral("%1").arg(++m_message_uid, 4, 10, QLatin1Char('0')) + "|";
158 QString hex_length = QString("%1").arg(string.length(), 2, 16, QLatin1Char('0'));
159 command.append(hex_length + "|" + string + "|");
160 command.append(QStringLiteral("%1").arg(Utils::crc16(Utils::str_data(command), command.length()),
161 4, 16, QLatin1Char('0')));
162 return command.toUtf8();
163 }
