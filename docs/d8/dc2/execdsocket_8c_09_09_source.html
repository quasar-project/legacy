78 void ExecdSocket::execute(QuasarSDK::Enums::NetworkCommand command) noexcept
79 {
80 QByteArray com;
81
82 switch(command)
83 {
84 case Enums::FormTelescopic:
85 this->setArgument("--remote", m_remote_address_list[0], Enums::Form);
86 com = Utils::wrapToExecdString(FROM_JSON("EXECD_FORM_TELESCOPIC") + m_args->formArgumentString(), &m_message_uid);
87 break;
88 case Enums::FocusImage:
89 this->setArgument("--remote", m_remote_address_list[0], Enums::Focus);
90 com = Utils::wrapToExecdString(FROM_JSON("EXECD_FOCUS_TELESCOPIC") + m_args->focusArgumentString(), &m_message_uid);
91 break;
92 case Enums::ReformImage:
93 this->setArgument("--remote", m_remote_address_list[0], Enums::Reform);
94 com = Utils::wrapToExecdString(FROM_JSON("EXECD_FORM_TELESCOPIC") + m_args->reformArgumentString(), &m_message_uid);
95 break;
96 case Enums::RemoteStorageStatus:
97 com = Utils::wrapToExecdString(FROM_JSON("EXECD_STORAGE_FETCH"), &m_message_uid);
98 break;
99 case Enums::ClearRemoteStorage:
100 com = Utils::wrapToExecdString(FROM_JSON("EXECD_STORAGE_CLEAR") + " " + FROM_JSON("EXECD_STORAGE_FETCH"), &m_message_uid);
101 break;
102 case Enums::StripStart:
103 {
104 this->setArgument("--remote", m_remote_address_list[0], Enums::Form);
105 if(m_compatibilityMode)
106 com = Utils::wrapToExecdString(FROM_JSON("EXECD_COMPAT_STRIP_START") + m_args->formArgumentString(), &m_message_uid);
107 else
108 {
109 QString strip_comma = FROM_JSON("EXECD_FORM_STRIP_START") + m_args->formArgumentString();
110 com = Utils::wrapToExecdString(condition(FROM_JSON("EXECD_SPECIAL_PID_OF") + "(strip_shot)",
111 FROM_JSON("EXECD_SPECIAL_PASS"), strip_comma), &m_message_uid);
112 m_strip_pid = m_message_uid;
113 }
114 break;
115 }
116 case Enums::StripStop:
117 this->setArgument("--remote", m_remote_address_list[0], Enums::Form);
118 if(m_compatibilityMode)
119 {
120 com = Utils::wrapToExecdString(FROM_JSON("EXECD_COMPAT_STRIP_STOP") + m_args->formArgumentString(), &m_message_uid);
121 break;
122 }
123 else
124 {
125 signalToProcess(m_strip_pid, Enums::SigINT);
126 return;
127 }
128 case Enums::StreamStart:
129 {
130 this->setArgument("--remote", m_remote_address_list[1], Enums::Form);
131 if(m_compatibilityMode)
132 return;
133 QString stream_comma = FROM_JSON("EXECD_FORM_STREAM_START") + m_args->formArgumentString();
134 com = Utils::wrapToExecdString(condition(FROM_JSON("EXECD_SPECIAL_PID_OF") + "(strip)",
135 FROM_JSON("EXECD_SPECIAL_PASS"), stream_comma), &m_message_uid);
136 m_strip_pid = m_message_uid;
137 break;
138 }
139 case Enums::StreamStop:
140 this->setArgument("--remote", m_remote_address_list[1], Enums::Form);
141 if(m_compatibilityMode)
142 return;
143 signalToProcess(m_strip_pid, Enums::SigINT);
144 return;
145 case Enums::Reboot:
146 {
147 com = Utils::wrapToExecdString(FROM_JSON("EXECD_REBOOT"), &m_message_uid);
148 break;
149 }
150 case Enums::PowerOff:
151 {
152 com = Utils::wrapToExecdString(FROM_JSON("EXECD_POWEROFF"), &m_message_uid);
153 break;
154 }
155 default:
156 qWarning() << "[EXECD] Incorrect command type";
157 return;
158 }
159
160 this->send(com);
161
162 qDebug().noquote() << "[EXECD] Sent built-in command";
163 emit metrics(com, com.length(), true);
164 }

211 void ExecdSocket::process(QByteArray data)
212 {
213 QString raw = data.data();
214 QString check_crc = raw;
215 check_crc.chop(5);
216
217 uint16_t crc16 = Utils::crc16(Utils::str_data(check_crc), check_crc.length());
218 uint16_t receivedCrc16 = raw.split("|").last().toUInt(nullptr, 16);
219
220 if(crc16 == receivedCrc16)
221 {
222 qDebug() << "[EXECD] Command executed successfully";
223 emit ping();
224 }
225// else
226// qWarning() << "[EXECD] CRC16 mismatch";
227 emit metrics(raw, data.size(), false);
228 }
