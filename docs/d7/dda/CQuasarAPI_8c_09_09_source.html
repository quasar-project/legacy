186 void QuasarAPI::setCurrentNetworkDelay(float o)
187 {
188 if(m_currentNetworkDelay == o)
189 return;
190 m_currentNetworkDelay = o;
191 emit currentNetworkDelayChanged();
192
193 if(m_currentNetworkDelay > Config::get()->value<float>("NETWORK_DELAY_THRESHOLD_DISCONNECT"))
194 setConnected(false);
195 else
196 setConnected(true);
197 }

310 void QuasarAPI::execute(QuasarSDK::Enums::NetworkCommand command) noexcept
311 {
312 switch(command)
313 {
314 case Enums::FormImage:
315 case Enums::SimpleStrip:
316 this->setArgument("--remote", m_remote_address_list[0], Enums::Form);
317 break;
318 case Enums::FocusImage:
319 this->setArgument("--remote", m_remote_address_list[0], Enums::Focus);
320 break;
321 case Enums::ReformImage:
322 this->setArgument("--remote", m_remote_address_list[0], Enums::Reform);
323 break;
324 case Enums::StartStrip:
325 case Enums::StopStrip:
326 this->setArgument("--remote", m_remote_address_list[1], Enums::Form);
327 break;
328 default: break;
329 }
330
331 execdSocket()->execute(command);
332 }
